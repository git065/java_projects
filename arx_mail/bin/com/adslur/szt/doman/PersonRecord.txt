package com.adslur.szt.domen;

public class PersonRecord {
	public String dmnd;
	public String msisdn;
	public String proezd;
	public String dom;

	public PersonRecord() {

	}

	public PersonRecord(String dmnd, String msisdn, String proezd, String dom) {

		this.dmnd = dmnd;
		this.msisdn = msisdn;
		this.proezd = proezd;
		this.dom = dom;

	}

}
